小程序图片上传至七牛云

小程序图片上传至七牛云这些好处什么的就不哔哩哔哩的说了…..
具体步骤如下：

导入qiniuUploader.js

首先下载七牛云qiniuUploader.js 在需要上传图片的js中文件中引入

const qiniuUploader = require("../../../utils/qiniuUploader.js")

与后台对接获取token

在onload函数中写请求，跟后端的接口对接，获取token

// 获取七牛云token
getToken() {
  let cookie = "jeesite.session.id=" + wx.getStorageSync("cookieKey");
  wx.request({
    url: config.basicUrl + "fault/oaFault/upToken",
    header: {
      Cookie: cookie
    },
    method: "POST",
    success: res => {
       this.setData({
          uploadToken: res.data
       })
     }
 })
},

图片选择途径

//图片选择
onChoise(event) {
  wx.showActionSheet({
    itemList: ["从相册中选择", "拍照", "取消"],
    success: res => {
      if (!res.cancel || res.tapIndex == 2) {
          if (res.tapIndex == 0) {
             this.chooseWxImage("album");
          } else if (res.tapIndex == 1) {
             this.chooseWxImage("camera");
          }
      }
   }
 });
}

选择图片并上传到七牛云

//图片选择途径并上传到七牛云
chooseWxImage(type) {
  wx.chooseImage({
    count: 1,
    sizeType: ["original", "compressed"],
    sourceType: [type],
    success: res => {
    const filepath = res.tempFilePaths[0]
    console.log('filepath', filepath)
    // 返回选定照片的本地文件路径列表，tempFilePath可以作为img标签的src属性显示图片
    qiniuUploader.upload(filepath, res => {  // 七牛云提供的api
    console.log('qiniuUploader.upload', res)
    this.setData({
    //注意这中间有个"/"
        tempFilePaths: 'http://pih3875s4.bkt.clouddn.com' + "/" + res.key
    })
        console.log('tempFilePaths', this.data.tempFilePaths)
    }, (error) => {
        console.log('error' + error)
    }, {
        region: 'ECN',                         // 华东区
        uploadURL: 'https://up-z1.qbox.me/',   // 所在大区的地址
        domain: 'pih3875s4.bkt.clouddn.com',   // 外链默认域名
        uptoken: this.data.uploadToken         // 后端返回来的uptoken
       })
    }
 });
},
}
